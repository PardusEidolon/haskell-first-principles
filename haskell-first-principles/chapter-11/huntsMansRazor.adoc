ePrinter :: Expr -> String
ePrinter (Lit n)   = show n
ePrinter (Add a b) = ePrinter a ++ " + " ++ ePrinter b

---
ePrinter (Add (Add (Lit 10) (Lit 20)) (Lit 40)) = ePrinter (Add (Lit 10) (Lit 20)) ++ " + " ++ ePrinter (Lit 40)
